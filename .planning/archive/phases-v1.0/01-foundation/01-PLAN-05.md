  <action>
Create the following files with exact specifications:

1. src/middleware/errorHandler.ts:
   - Import { Request, Response, NextFunction } from 'express'
   - Import { ZodError } from 'zod'
   - Import { Prisma } from '@prisma/client'
   - Export function errorHandler(err: Error, req: Request, res: Response, _next: NextFunction): void
     - Log error to console: console.error(`[ERROR] ${req.method} ${req.path}:`, err.message)
     - If err instanceof ZodError:
       - res.status(400).json({ success: false, error: { code: "VALIDATION_ERROR", message: "Request validation failed", details: err.errors.map(e => ({ path: e.path.join('.'), message: e.message })) } })
     - Else if err instanceof Prisma.PrismaClientKnownRequestError:
       - If code === 'P2002' (unique constraint): res.status(409).json({ success: false, error: { code: "DUPLICATE_ERROR", message: "Resource already exists" } })
       - If code === 'P2025' (record not found): res.status(404).json({ success: false, error: { code: "NOT_FOUND", message: "Resource not found" } })
       - Else: res.status(500).json({ success: false, error: { code: "DATABASE_ERROR", message: "Database operation failed" } })
     - Else if err.message:
       - res.status(400).json({ success: false, error: { code: "BAD_REQUEST", message: err.message } })
     - Else:
       - res.status(500).json({ success: false, error: { code: "INTERNAL_ERROR", message: "Internal server error" } })

2. src/middleware/rateLimiter.ts:
   - Install express-rate-limit if not present: npm install express-rate-limit
   - Import rateLimit from 'express-rate-limit'
   - Export const authRateLimiter = rateLimit({
       windowMs: 15 * 60 * 1000, // 15 minutes
       max: 5, // 5 requests per window
       message: { success: false, error: { code: "RATE_LIMITED", message: "Too many attempts. Please try again later." } },
       standardHeaders: true,
       legacyHeaders: false,
     })

Note: The rate limiter should be applied ONLY to auth routes (register/login), not to all routes.
